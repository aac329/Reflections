    function attachmentClickBind() {
        $('#btnAttachments').on("click", function (event) {
            if ($('#upload_control').is(":hidden")) {
                $('#upload_control').slideDown("slow");
                event.preventDefault();
                $('#btnAttachments').html('Close Attachments');
            } else {
                $("#upload_control").slideUp("slow");
                event.preventDefault();
                $('#btnAttachments').html('Open Attachments');
            }
        });
    }
    //Attempt to keep the attachment control open or closed through postbacks
    function SaveState(sender, args) {
        upload_control_state = $('#upload_control').css("display");
        upload_button_state = $('#btnAttachments').html();
    }

    function ReapplyState(sender, args) {
        $('#upload_control').css("display", upload_control_state);
        $('#btnAttachments').html(upload_button_state);
        attachmentClickBind();
    }

    function openProjects() {
        var loc1 = $('#txtProjects').val();
        if (loc1 != "")
            window.open("file://" + loc1);
    }
