//This needs a method to load the Programs from ManageDB.cs

protected void LoadPrograms()
{
	var liItems = String.Empty;
	var dbo = new ManageDB();
	foreach (DataRow row in dbo.getPrograms().Rows)
	{
		var item = String.Format("<li><a href='tme/Programs.aspx?id={0}'>{1}</a></li>", row["Program_ID"], row["Title"]);
		liItems += item;
	}

	ltPrograms.Text = liItems;
}






(\Tme\App_Code\DAL\ManageDB.cs)
//This method will get the actual raw Data from [Distance].[dbo].[Tme_Program]

        public DataTable getPrograms()
        {
            var dt = new DataTable();
	        using (var cn = new SqlConnection(Connection))
	        {
		        try
		        {
			        string sql = @"select distinct Program_ID, Title
					        from Distance.dbo.Tme_Program";
			
			        var da = new SqlDataAdapter(sql, cn);

			        da.Fill(dt);
		        }
		        catch (SqlException e)
		        {
			        HttpContext.Current.Trace.Warn("Sql Error", e.ToString());
		        }
		        return dt;
	        }
        }
