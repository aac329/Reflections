<%@ Control Language="C#" AutoEventWireup="true" CodeBehind="ProgramSearch.ascx.cs" Inherits="Tme.Controls.UserControls.ProgramSearch" %>

<script type ="text/javascript" >
    function GetRadWindowProgSearch() {
        var oWindow = null;
        if (window.radWindow)
            oWindow = window.RadWindow;
        else if (window.frameElement.radWindow)
            oWindow = window.frameElement.radWindow;
        return oWindow;
    }

    function CloseProgSearch() {
        GetRadWindowProgSearch().Close();
    }

    function redirectpageProgSearch() {
        var programText = $('.txtProgramClass').val();
        var url = "/tme/ProgramSearch.aspx?title=" + programText + "&status=" + $('.rblTmeProgram input:checked').val();
        GetRadWindowProgSearch().BrowserWindow.location.href = url;
    }
</script>

</div>
